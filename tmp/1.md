```sql
CREATE EXTENSION pg_trgm;


CREATE DATABASE musicbrainz;

\c musicbrainz

CREATE TABLE albums (
    id SERIAL PRIMARY KEY,
    release_id VARCHAR(255) NOT NULL DEFAULT '',
    catalognumber VARCHAR(255) NOT NULL DEFAULT '',
    date VARCHAR(255) NOT NULL DEFAULT '',
    album TEXT NOT NULL DEFAULT '',
    tracks TEXT NOT NULL DEFAULT '',
    themes TEXT[] NOT NULL DEFAULT '{}',
    links TEXT[] NOT NULL DEFAULT '{}'
);

-- 添加索引
CREATE INDEX idx_albums_release_id ON albums(release_id);
CREATE INDEX idx_albums_album_trgm ON albums USING gin (album gin_trgm_ops);
CREATE INDEX idx_albums_catalognumber_trgm ON albums USING gin (catalognumber gin_trgm_ops);
CREATE INDEX idx_albums_tracks_trgm ON albums USING gin (tracks gin_trgm_ops);


COPY albums (release_id, catalognumber, date, album, tracks, themes, links)
FROM 'D:\\1.csv' 
WITH (
    FORMAT csv,
    HEADER true,
    DELIMITER ',',
    QUOTE '"',
    NULL '',
    ENCODING 'UTF-8'
);


SELECT count(*) FROM albums;

SELECT count(*) FROM albums;

DROP TABLE albums;

SELECT * FROM albums
WHERE random() < 0.01
LIMIT 10;
```

## 搜索

```sh
\x on
\timing on

SELECT *,
       (
         similarity(catalognumber, 'COBC-5991') +
         similarity(album, 'THE IDOLM@STER 5th ANNIVERSARY The world is all one !! 100704 Disc 2 [COBC-5991]')  + 
         similarity(tracks, 'xxxx')
       ) AS match_score
FROM albums
ORDER BY match_score DESC
LIMIT 50;
```
